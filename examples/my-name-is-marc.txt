section .data
    msg db "Hey my name is Marc", 0xA   
    len equ $ - msg                    

section .text
    global _start

_start:
    ; write(stdout, msg, len)
    mov eax, 4         
    mov ebx, 1        
    mov ecx, msg        
    mov edx, len      
    int 0x80          

    ; exit(0)
    mov eax, 1         
    xor ebx, ebx        
    int 0x80
